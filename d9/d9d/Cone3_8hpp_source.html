109  RayIntersection3 rayIntersection(Ray3 const & ray, Real max_time = -1) const
110  {
111  int surface;
112  Real t = rayIntersectionTime(ray, max_time, surface);
113  if (surface >= 0)
114  {
115  Vector3 n;
116  switch (surface)
117  {
118  case 0: n = -unit_dir; break;
119  case 1:
120  {
121  Vector3 p = ray.getPoint(t);
122  Vector3 cp = segment.closestPoint(p);
123  Vector3 h = getApex() - cp;
124  Vector3 d = p - cp;
125  Real hlen2 = h.squaredNorm();
126  Real dlen2 = d.squaredNorm();
127  if (dlen2 > 0)
128  n = (h + (hlen2 / dlen2) * d).stableNormalized(); // Exchange lengths of h and d. FIXME: Avoid the final sqrt?
129  else
130  n = Vector3::Zero();
131  break;
132  }
133  default: n = Vector3::Zero();
134  }
135 
136  return RayIntersection3(t, &n);
137  }
138  else
139  return RayIntersection3(-1);
140  }

145  {
146  len = segment.length();
147  if (len > 0 && base_radius > 0)

165  Real rayIntersectionTime(Ray3 const & ray, Real max_time, int & surface) const
166  {
167  if (contains(ray.getOrigin()))
168  {
169  surface = 2;
170  return 0;
171  }
172 
173  surface = -1;
174  Ray3 local_ray = ray.transform(world2local);
175  double min_root = -1;
176 
177  // Base
178  Plane3 plane = Plane3::fromPointAndNormal(Vector3::Zero(), Vector3::UnitZ());
179  auto plane_time = plane.rayIntersectionTime(local_ray, max_time);
180  if (plane_time >= 0 && (min_root < 0 || plane_time < min_root))
181  {
182  Vector3 p = local_ray.getPoint(plane_time);
183  if (p.head(2).squaredNorm() <= 1)
184  {
185  min_root = plane_time;
186  surface = 0;
187  }
188  }
189 
190  // Sides
191  Vector3 p2 = local_ray.getOrigin().cwiseProduct(local_ray.getOrigin());
192  Vector3 pu = local_ray.getOrigin().cwiseProduct(local_ray.getDirection());
193  Vector3 u2 = local_ray.getDirection().cwiseProduct(local_ray.getDirection());
194  Real diff_z = local_ray.getOrigin().z() - 1;
195 
196  double s[3];
197  s[2] = u2[0] + u2[1] - u2[2];
198  s[1] = 2 * (pu[0] + pu[1] - diff_z * local_ray.getDirection()[2]);
199  s[0] = p2[0] + p2[1] - diff_z * diff_z;
200 
201  double roots[2];
202  int num_roots = Math::solveQuadratic(s[0], s[1], s[2], roots);
203 
204  for (int i = 0; i < num_roots; ++i)
205  if (roots[i] >= 0 && (max_time < 0 || roots[i] <= max_time) && (min_root < 0 || roots[i] < min_root))
206  {
207  Vector3 point = local_ray.getPoint((Real)roots[i]);
208  if (point[2] >= 0 && point[2] <= 1)
209  {
210  min_root = roots[i];
211  surface = 1;
212  }
213  }
214 
215  return (Real)min_root;
216  }
