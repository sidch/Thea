611  template <typename InputIterator>
612  void init(InputIterator begin, InputIterator end, Method method = Method::AUTO, intx max_depth_ = -1,
613  intx max_elems_per_leaf_ = -1, bool save_memory = false, bool deallocate_previous_memory = true)
614  {
615  if (method == Method::AUTO)
616  {
617  if (MaxDegree >= 4 && N == 2)
618  method = Method::QUADTREE;
619  else if (MaxDegree >= 8 && N == 3)
620  method = Method::OCTREE;
621  else if (MaxDegree >= 2)
622  method = Method::KDTREE;
623  else
624  throw Error("BvhN: Could not infer tree construction method");
625  }
626 
627  clear(deallocate_previous_memory);
628 
629  if (deallocate_previous_memory)
630  {
631  for (InputIterator ii = begin; ii != end; ++ii, ++num_elems)
632  if (elementPassesFilters(*ii))
633  elems.push_back(*ii);
634  }
635  else
636  {
637  size_t max_new_elems = (size_t)std::distance(begin, end);
638  bool resized = false;
639  if (max_new_elems > elems.size())
640  {
641  if (filters.empty())
642  elems.resize((size_t)std::ceil(1.2 * max_new_elems)); // add a little more space to avoid future reallocs
643  else
644  elems.clear(); // we don't know how many elements will pass the filter
645 
646  resized = true;
647  }
648 
649  if (filters.empty())
650  {
651  std::copy(begin, end, elems.begin());
652  num_elems = (intx)max_new_elems;
653  }
654  else
655  {
656  if (resized)
657  {
658  for (InputIterator ii = begin; ii != end; ++ii)
659  if (elementPassesFilters(*ii))
660  {
661  elems.push_back(*ii);
662  ++num_elems;
663  }
664  }
665  else
666  {
667  typename ElementArray::iterator ei = elems.begin();
668  for (InputIterator ii = begin; ii != end; ++ii)
669  if (elementPassesFilters(*ii))
670  {
671  *(ei++) = *ii;
672  ++num_elems;
673  }
674  }
675  }
676  }
677 
678  if (num_elems <= 0)
679  return;
680 
681  switch (method)
682  {
683  case Method::KDTREE:
684  initKdTree(begin, end, max_depth_, max_elems_per_leaf_, save_memory, deallocate_previous_memory); break;
685 
686  default: throw Error("BvhN: Unsupported BVH construction method");
687  }

1305  // the node's indices will be deallocated from the main index pool (and possibly moved to the leaf pool).

1317 #ifdef THEA_BVHN_KDTREE_SPLIT_LONGEST

1319 #else

1321 #endif
