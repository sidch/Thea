37 // Called for each point in the neighborhood.
38 struct ProjectedCurvatureFunctor : private Noncopyable
39 {
40  ProjectedCurvatureFunctor(Vector3 const & p, Vector3 const & n)
41  : position(p), normal(n), num_offsets(0), sum_offsets(Vector3::Zero())
42  {}
43 
44  template <typename SampleT> bool operator()(intx index, SampleT const & t)
45  {
46  if (NormalTraitsN<SampleT, 3>::getNormal(t).dot(normal) > -1.0e-05f) // ignore points on hidden side
47  {
48  Vector3 offset = (PointTraitsN<SampleT, 3>::getPosition(t) - position).stableNormalized();
49  sum_offsets += offset;
50  num_offsets++;
51  }
52 
53  return false;
54  }
55 
56  double getCurvature() const
57  {
58  return num_offsets > 0 ? -sum_offsets.dot(normal) / num_offsets : 0;
59  }
60 
61  Vector3 position, normal;
62  intx num_offsets;
63  Vector3 sum_offsets;
64 
65 }; // struct ProjectedCurvatureFunctor

83 Curvature::computeProjectedCurvature(Vector3 const & position, Vector3 const & normal, Real nbd_radius) const
84 {
85  if (nbd_radius < 0)
86  nbd_radius = 0.1f;
87 
88  nbd_radius *= surf->getScale();
89 
90  CurvatureInternal::ProjectedCurvatureFunctor func(position, normal);
91  Ball3 range(position, nbd_radius);
92  const_cast<PointSet3::SampleBvh &>(surf->getBvh()).processRangeUntil<IntersectionTester>(range, std::ref(func));
93 
94  return func.getCurvature();
