27 intx
28 midpoint(size_t i, size_t j, Array<Vector3> & vertices, MidpointMap & midpoints)
29 {
30  IndexPair edge = (i < j ? IndexPair(i, j) : IndexPair(j, i));
31  MidpointMap::const_iterator existing = midpoints.find(edge);
32  if (existing == midpoints.end())
33  {
34  Vector3 p = (vertices[i] + vertices[j]).normalized();
35  midpoints[edge] = vertices.size();
36  vertices.push_back(p);
37 
38  return vertices.size() - 1;
39  }
40  else
41  return existing->second;
42 }

74 {
75  // http://www.opengl.org.ru/docs/pg/0208.html
76 
77  if (num_subdivs < 0)

79  THEA_ERROR << "GeodesicSphere3: Number of subdivisions must be non-negative";

83  static Real const ICO_X = 0.525731112119133606;
84  static Real const ICO_Z = 0.850650808352039932;
85 
86  static Vector3 const ICO_VERTS[12] = {
87  Vector3(-ICO_X, 0.0, ICO_Z), Vector3(ICO_X, 0.0, ICO_Z),
88  Vector3(-ICO_X, 0.0, -ICO_Z), Vector3(ICO_X, 0.0, -ICO_Z),
89 
90  Vector3(0.0, ICO_Z, ICO_X), Vector3(0.0, ICO_Z, -ICO_X),
91  Vector3(0.0, -ICO_Z, ICO_X), Vector3(0.0, -ICO_Z, -ICO_X),
92 
93  Vector3(ICO_Z, ICO_X, 0.0), Vector3(-ICO_Z, ICO_X, 0.0),
94  Vector3(ICO_Z, -ICO_X, 0.0), Vector3(-ICO_Z, -ICO_X, 0.0)
95  };
96 
97  static size_t const ICO_TRIS[20][3] = {
98  { 1, 4, 0}, { 4, 9, 0}, {4, 5, 9}, {8, 5, 4}, { 1, 8, 4},
99  { 1, 10, 8}, {10, 3, 8}, {8, 3, 5}, {3, 2, 5}, { 3, 7, 2},
100  { 3, 10, 7}, {10, 6, 7}, {6, 11, 7}, {6, 0, 11}, { 6, 1, 0},
101  {10, 1, 6}, {11, 0, 9}, {2, 11, 9}, {5, 2, 9}, {11, 2, 7}
102  };
103 
104  vertices.resize(12);
105  for (size_t i = 0; i < 12; ++i)
106  vertices[i] = ICO_VERTS[i];
107 
108  if (num_subdivs == 0 && triangles)
109  {
110  triangles->resize(20 * 3);
111  for (size_t i = 0; i < 20; ++i)
112  {
113  (*triangles)[3 * i ] = (intx)ICO_TRIS[i][0];
114  (*triangles)[3 * i + 1] = (intx)ICO_TRIS[i][1];
115  (*triangles)[3 * i + 2] = (intx)ICO_TRIS[i][2];
116  }
117  }
118 
119  if (num_subdivs > 0)
120  {
121  if (triangles)
122  triangles->clear();
123 
124  GeodesicSphere3Internal::MidpointMap midpoints;
125  for (size_t i = 0; i < 20; ++i)
126  {
127  GeodesicSphere3Internal::geodesicSphereSubdivide(num_subdivs, ICO_TRIS[i][0], ICO_TRIS[i][1], ICO_TRIS[i][2],
128  vertices, triangles, midpoints);
129  }
130  }
131 
132  return true;
133 }
