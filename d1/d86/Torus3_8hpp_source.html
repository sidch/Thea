135  Real rayIntersectionTime(Ray3 const & ray, Real max_time, Vector3 * normal) const
136  {
137  if (contains(ray.getOrigin()))
138  {
139  if (normal) *normal = Vector3::Zero();
140  return 0;
141  }
142 
143  Ray3 local_ray = ray.transform(world2local);
144  double min_root = -1;
145 
146  double r2pw2 = outer_radius * outer_radius + inner_radius * inner_radius;
147  double r2mw2 = r2pw2 - 2 * inner_radius * inner_radius;
148 
149  Vector3 p2 = local_ray.getOrigin().cwiseProduct(local_ray.getOrigin());
150  Vector3 pu = local_ray.getOrigin().cwiseProduct(local_ray.getDirection());
151  Vector3 u2 = local_ray.getDirection().cwiseProduct(local_ray.getDirection());
152 
153  double s[5];
154  s[4] = u2[0] * (u2[0] + 2 * u2[1]) + u2[1] * (u2[1] + 2 * u2[2]) + u2[2] * (u2[2] + 2 * u2[0]);
155  s[3] = 4 * (pu[0] + pu[1] + pu[2]) * (u2[0] + u2[1] + u2[2]);
156  s[2] = 2 * (r2mw2 * u2[2] - r2pw2 * (u2[0] + u2[1]))
157  + 8 * (pu[0] * pu[1] + pu[1] * pu[2] + pu[2] * pu[0])
158  + 6 * (pu[0] * pu[0] + pu[1] * pu[1] + pu[2] * pu[2])
159  + 2 * (p2[0] * (u2[1] + u2[2]) + p2[1] * (u2[2] + u2[0]) + p2[2] * (u2[0] + u2[1]));
160  s[1] = 4 * (r2mw2 * pu[2] - r2pw2 * (pu[0] + pu[1]) + (p2[0] + p2[1] + p2[2]) * (pu[0] + pu[1] + pu[2]));
161  s[0] = 2 * (r2mw2 * p2[2] - r2pw2 * (p2[0] + p2[1]) + p2[0] * p2[1] + p2[1] * p2[2] + p2[2] * p2[0])
162  + p2[0] * p2[0] + p2[1] * p2[1] + p2[2] * p2[2] + r2mw2 * r2mw2;
163 
164  double roots[4];
165  int num_roots = Math::solveQuartic(s[0], s[1], s[2], s[3], s[4], roots);
166 
167  for (int i = 0; i < num_roots; ++i)
168  if (roots[i] >= 0 && (max_time < 0 || roots[i] <= max_time) && (min_root < 0 || roots[i] < min_root))
169  min_root = roots[i];
170 
171  if (normal && min_root >= 0)
172  {
173  // Exploit orthonormality of world2local
174  Vector3 local_p = local_ray.getPoint((Real)min_root);
175  Vector2 ring_point = outer_radius * local_p.head<2>().stableNormalized();
176  *normal = (world2local.getLinear() * (local_p - Vector3(ring_point.x(), ring_point.y(), 0))).stableNormalized();
177  }
178 
179  return (Real)min_root;
180  }
