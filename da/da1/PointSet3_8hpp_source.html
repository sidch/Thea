321  template <typename RayQueryStructureT> void updateGraph(RayQueryStructureT const * surface) const
322  {
323  if (valid_graph) // don't initialize twice
324  return;
325 
326  SampleBvh * h = nullptr;
327  SampleBvh agg_bvh;
328  if (!dense_samples.empty())
329  {
330  // Aggregate samples
331  Array<Sample *> sample_ptrs(samples.size() + dense_samples.size());
332 
333  for (size_t i = 0; i < samples.size(); ++i)
334  sample_ptrs[i] = const_cast<Sample *>(&samples[i]);
335 
336  for (size_t i = 0; i < dense_samples.size(); ++i)
337  {
338  auto & ds = const_cast<Sample &>(dense_samples[i]);
339  ds.setIndex((intx)(samples.size() + i)); // indices of dense samples don't matter and can be changed internally
340  sample_ptrs[samples.size() + i] = &ds;
341  }
342 
343  // Clear any prior adjacency data
344  for (size_t i = 0; i < sample_ptrs.size(); ++i)
345  sample_ptrs[i]->clearNeighbors();
346 
347  if (sample_ptrs.size() <= 1) // nothing to do
348  return;
349 
350  // Create BVH on aggregated samples
351  agg_bvh.init(sample_ptrs.begin(), sample_ptrs.end());
352  h = &agg_bvh;
353  }
354  else
355  h = const_cast<SampleBvh *>(&getBvh());
356 
357  // Get a measure of the average pairwise separation of samples
358  auto h_num_elems = h->numElements();
359  auto h_elems = h->getElements();
360  avg_separation = 0;
361  intx num_trials = std::min(100L, h_num_elems);
362  for (intx i = 0; i < num_trials; ++i)
363  {
364  auto index = Random::common().integer(0, (int32)h_num_elems - 1);
365  FilterSelf filter(h_elems[index]);
366  h->pushFilter(&filter);
367  intx nn_index = h->closestElement<MetricL2>(h_elems[index]->getPosition());
368  alwaysAssertM(nn_index >= 0, "PointSet3: Nearest neighbor of sample not found");
369  avg_separation += (h_elems[nn_index]->getPosition() - h_elems[index]->getPosition()).squaredNorm();
370  h->popFilter();
371  }
372 
373  avg_separation = std::sqrt(avg_separation / num_trials); // RMS average
374 
375  // Find the neighbors of each sample
376  Real sep_scale = std::sqrt((Real)options.max_degree); // assume uniform distribution on 2D surface
377  for (intx i = 0; i < h_num_elems; ++i)
378  findSampleNeighbors(h_elems[i], *h, sep_scale * avg_separation, surface);
379 
380  // Extract adjacencies between original set of samples
381  if (!dense_samples.empty())
382  extractOriginalAdjacencies(h_num_elems, h_elems);
383 
384  // Update the average separation to the correct value
385  updateAverageSeparation();
386 
387  valid_graph = true;
388  }

525  template <typename TriangleT> Vector3 smoothNormal(TriangleT const & tri, Vector3 const & p)
526  {

583  template <typename RayQueryStructureT> class NeighborFunctor
584  {
585  public:
587  NeighborFunctor(Sample * sample_, bool has_normals_, RayQueryStructureT const * surface_)
588  : sample(sample_), has_normals(has_normals_), surface(surface_)
589  {
590  debugAssertM(sample, "PointSet3: Can't create neighbor functor without valid source sample");
591  }
592 
594  bool operator()(intx index, Sample * nbr)
595  {
596  Real sep = 0;
597  if (isValidNeighbor(nbr, sep))
598  sample->addNeighbor(Sample::Neighbor(nbr, sep));
599 
600  return false;
601  }
602 
603  private:
609  bool isValidNeighbor(Sample const * nbr, Real & sep)
610  {
611  // Tests are (hopefully) in decreasing order of speed
612 
613  // Identity test
614  if (sample == nbr)
615  return false;
616 
617  // Angle test
618  if (has_normals)
619  {
620  static Real const MIN_DOT = -0.5f;
621  if (sample->getNormal().dot(nbr->getNormal()) < MIN_DOT)
622  return false;
623  }
624 
625  // Duplication test which compares only the samples themselves (since numerical error can cause multiple attempts to
626  // insert the same sample with slightly different separations)
627  for (int i = 0; i < sample->numNeighbors(); ++i)
628  if (nbr == sample->getNeighbor(i).getSample())
629  return false;
630 
631  // Compute the separation of the samples
632  Vector3 diff = nbr->getPosition() - sample->getPosition();
633  Real sqsep = diff.squaredNorm();
634  sep = Math::fastSqrt(sqsep);
635 
636  // Reachability test: lift points off the surface by an amount proportional to their separation, and see if the line
637  // connecting them is blocked by the surface
638  if (surface && has_normals)
639  {
640  static Real const LIFT_FACTOR = 5;
641  Vector3 lift_dir = (sample->getNormal() + nbr->getNormal()).normalized();
642  typename RayQueryStructureT::RayT ray(sample->getPosition() + LIFT_FACTOR * sep * lift_dir, diff);
643  if (surface->template rayIntersects<RayIntersectionTester>(ray, 1))
644  return false;
645  }
646 
647  return true;
648  }
649 
650  Sample * sample;
651  bool has_normals;
652  RayQueryStructureT const * surface;
653 
654  }; // struct NeighborFunctor

657  template <typename RayQueryStructureT>
658  void findSampleNeighbors(Sample * sample, SampleBvh & h, Real init_radius, RayQueryStructureT const * surface) const
659  {
660  static int const MAX_ITERS = 3;
661  static Real const RADIUS_EXPANSION_FACTOR = 2.0f;
662  int min_degree = Math::clamp((int)(0.25 * options.max_degree), 4, options.max_degree);
663 
664  Real radius = init_radius;
665  for (int i = 0; i < MAX_ITERS; ++i)
666  {
667  sample->clearNeighbors();
668  NeighborFunctor<RayQueryStructureT> func(sample, has_normals, surface);
669 
670  Ball3 nbd(sample->getPosition(), radius);
671  h.processRangeUntil<IntersectionTester>(nbd, func);
672 
673  if (sample->numNeighbors() >= min_degree)
674  break;
675  else
676  radius *= RADIUS_EXPANSION_FACTOR;
677  }
678  }

686  void updateAverageSeparation() const
687  {
688  avg_separation = 0;
689  intx num_edges = 0; // double-counts, but we'll ignore that for now
690  for (size_t i = 0; i < samples.size(); ++i)
691  {
692  for (auto const & n : samples[i].getNeighbors())
693  avg_separation += n.getSeparation();
