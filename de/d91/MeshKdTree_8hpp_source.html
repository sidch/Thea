117  void updateElements(typename BaseT::Node const * start = nullptr)
118  {
119  // Recompute triangle cached properties in case vertex positions have changed
120  if (!start || start == this->getRoot()) // non-recursive loop avoids double-processing elements spanning multiple leaves
121  {
122  auto const * all_elems = this->getElements();
123  for (auto ei = all_elems, all_elems_end = all_elems + BaseT::numElements(); ei != all_elems_end; ++ei)
124  ei->update();
125  }
126  else
127  updateElementsRecursive(start);
128  }

144  for (auto ei = start->elementIndicesBegin(), ei_end = start->elementIndicesEnd(); ei != ei_end; ++ei)
