123  void updateElements(typename BaseT::Node const * start = nullptr)
124  {
125  // Recompute triangle cached properties in case vertex positions have changed
126  if (!start || start == this->getRoot()) // non-recursive loop avoids double-processing elements spanning multiple leaves
127  {
128  auto const * all_elems = this->getElements();
129  for (auto ei = all_elems, all_elems_end = all_elems + BaseT::numElements(); ei != all_elems_end; ++ei)
130  ei->update();
131  }
132  else
133  updateElementsRecursive(start);
134  }

143  private:
145  void updateElementsRecursive(typename BaseT::Node const * start)
146  {
147  if (start->isLeaf())
148  {
149  auto const * all_elems = BaseT::getElements();
150  for (auto ei = start->elementIndicesBegin(), ei_end = start->elementIndicesEnd(); ei != ei_end; ++ei)
151  all_elems[*ei].update();
152  }
153  else
154  {
155  for (intx i = 0; i < start->maxDegree(); ++i)
156  {
157  auto c = start->getChild(i);
158  if (c) { updateElementsRecursive(c); }
159  }
160  }
