828 inline uint32
829 DisplayMeshFace::getVertexIndex(int i) const
830 {
831  debugAssertM(i >= 0 && i < num_vertices, "DisplayMeshFace: Vertex index out of bounds");
832 
833  switch (num_vertices)
834  {
835  case 3:
836  case 4: return mesh->tris[3 * first_tri + (i < 3 ? i : 4)]; // exploit vertex ordering guarantee of triangulateQuad()
837 
838  default:
839  {
840  auto loc = mesh->large_poly_verts.find(first_tri);
841  debugAssertM(loc != mesh->large_poly_verts.end(), "DisplayMeshFace: Vertices of high-degree polygon not found");
842  return loc->second[(size_t)i];
843  }
844  }
