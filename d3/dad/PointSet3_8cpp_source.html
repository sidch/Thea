31 void
32 updateNeighborPointers(PointSet3::SampleArray & samples, PointSet3::SampleArray const & src_samples)
33 {
34  alwaysAssertM(samples.size() == src_samples.size(),
35  "PointSet3: Can't update sample neighbor pointers from source array of different size");
36 
37  for (size_t i = 0; i < samples.size(); ++i)
38  {
39  auto & nbrs = samples[i].getNeighbors();
40  for (int j = 0; j < nbrs.size(); ++j)
41  {
42  size_t index = nbrs[j].getSample() - &src_samples[0]; // take advantage of array storage (this is NOT getIndex())
43  debugAssertM(index >= 0 && index < samples.size(), "PointSet3: Can't get array index of neighboring sample");
44 
45  // Again, because of array storage, this should not break the relative ordering of neighbors with equal separation, since
46  // pointer less-than is preserved
47  const_cast<PointSet3::Sample::Neighbor &>(nbrs[j]).setSample(&samples[index]);
48  }
49  }
50 }

103 
104  VertexIterator verticesBegin() { return nodes; }
105  VertexConstIterator verticesBegin() const { return nodes; }

132 // Callback for shortest paths algorithm.
133 struct DijkstraCallback
134 {
135  DijkstraCallback(PointSet3::Sample * sample_, intx num_orig_samples_, int max_nbrs_)
136  : sample(sample_), num_orig_samples(num_orig_samples_), max_nbrs(max_nbrs_)
137  {
138  sample->getNeighbors().clear();
139  }
140 
141  bool operator()(SamplePointerGraph::VertexHandle vertex, double distance, bool has_pred,
142  SamplePointerGraph::VertexHandle pred)
143  {
144  // Assume original samples are at the head of the list
145  if (vertex->getIndex() != sample->getIndex() && vertex->getIndex() < num_orig_samples)
146  sample->getNeighbors().insert(PointSet3::Sample::Neighbor(vertex, (Real)distance));
147 
148  return sample->getNeighbors().size() >= max_nbrs;
149  }
150 
151  PointSet3::Sample * sample;
152  intx num_orig_samples;
153  int max_nbrs;
154 };

158 void
159 PointSet3::extractOriginalAdjacencies(intx num_samples, Sample ** sample_ptrs) const
160 {
161  using namespace PointSet3Internal;
162 
163  SamplePointerGraph graph(num_samples, sample_ptrs);
164  ShortestPaths<SamplePointerGraph> shortest_paths;
165 
166  // The graph is considered a mutable property of the point set so the const_casts below are ok because only neighbors are
167  // modified
168  SampleArray samples_with_new_nbrs(samples.size());
169  for (size_t i = 0; i < samples.size(); ++i)
170  {
171  samples_with_new_nbrs[i] = samples[i];
172  DijkstraCallback callback(&samples_with_new_nbrs[i], (intx)samples.size(), options.max_degree);
173  shortest_paths.dijkstraWithCallback(graph, const_cast<Sample *>(&samples[i]), callback);
174  }
175 
176  for (size_t i = 0; i < samples.size(); ++i)
177  const_cast<Sample &>(samples[i]).getNeighbors() = samples_with_new_nbrs[i].getNeighbors();
178 }

180 bool
181 PointSet3::load(std::string const & samples_path, std::string const & graph_path)
182 {
183  clear();
184 
185  // Load samples
186  std::ifstream sin(samples_path.c_str());
187  if (!sin)
188  {
189  THEA_ERROR << "PointSet3: Could not open samples file '" << samples_path << "' for reading";
190  return false;
191  }
192 
193  has_normals = false;
194 
195  std::string line;
196  Vector3 p, n;
197  while (std::getline(sin, line))
198  {
199  std::istringstream line_in(line);
200  if (!(line_in >> p[0] >> p[1] >> p[2]))
201  {
202  THEA_ERROR << "Could not read sample " << samples.size() << " from '" << samples_path << '\'';
203  return false;
204  }
205 
206  bool sample_has_normal = (bool)(line_in >> n[0] >> n[1] >> n[2]);
207  if (samples.empty())
208  has_normals = sample_has_normal;
209  else if (has_normals != sample_has_normal)
210  {
211  THEA_ERROR << "PointSet3: Some samples in '" << samples_path << "' have normals and some don't";
212  return false;
213  }
214 
215  samples.push_back(Sample((intx)samples.size(), 0));
216  samples.back().setPosition(p);
217  if (has_normals) samples.back().setNormal(n);
218  }
219 
220  invalidateAll(); // graph, scale etc must be recomputed
221 
222  // Load graph, if a non-empty path was specified. Else, it must be computed with updateGraph().
223  if (graph_path.empty())
224  return true;
225 
226  std::ifstream gin(graph_path.c_str());
227  if (!gin)
228  {
229  THEA_ERROR << "PointSet3: Could not open graph file '" << graph_path << "' for reading";
230  return false;
231  }
232 
233  if (std::getline(gin, line))
234  {
235  std::istringstream line_in(line);
236  intx max_nbrs;
237  if (!(line_in >> max_nbrs) || max_nbrs < 0)
238  {
239  THEA_ERROR << "PointSet3: Could not read valid maximum degree from '" << graph_path << '\'';
240  return false;
241  }
242 
243  options.setMaxDegree(max_nbrs);
244  }
245  else
246  {
247  THEA_ERROR << "PointSet3: Could not read maximum degree from '" << graph_path << '\'';
248  return false;
249  }
250 
251  Array<Sample::Neighbor> nbrs((size_t)options.max_degree);
252  intx num_nbrs, nbr_index;
253  for (size_t i = 0; i < samples.size(); ++i)
254  {
255  if (!std::getline(gin, line))
256  {
257  THEA_ERROR << "PointSet3: Could not read neighbors of vertex " << i << " from '" << graph_path << '\'';
258  return false;
259  }
260 
261  std::istringstream line_in(line);
262  if (!(line_in >> num_nbrs) || num_nbrs < 0)
263  {
264  THEA_ERROR << "PointSet3: Could not read valid degree of vertex " << i << " from '" << graph_path << '\'';
265  return false;
266  }
267 
268  for (intx j = 0; j < num_nbrs; ++j)
269  {
270  if (!(line_in >> nbr_index) || nbr_index < 0 || nbr_index >= (intx)samples.size())
271  {
272  THEA_ERROR << "PointSet3: Could not read valid neighbor " << j << " of vertex " << i << " from '" << graph_path
273  << '\'';
274  return false;
275  }
276 
277  Sample * nbr_sample = &samples[(size_t)nbr_index];
278  nbrs[(size_t)j] = Sample::Neighbor(nbr_sample, -1);
279  }
280 
281  // Check if precomputed separations exist, else compute separations as Euclidean distances
282  Real nbr_sep;
283  if (line_in >> nbr_sep)
284  {
285  for (size_t j = 0; j < nbrs.size(); ++j)
286  {
287  if (j > 0)
288  {
289  if (!(line_in >> nbr_sep))
290  {
291  THEA_ERROR << "PointSet3: Could not read separation of neighbor " << j << " of vertex " << i << " from '"
292  << graph_path << '\'';
293  return false;
294  }
295  }
296 
297  nbrs[j].setSeparation(nbr_sep);
298  }
299  }
300  else
301  {
302  for (size_t j = 0; j < nbrs.size(); ++j)
303  {
304  nbr_sep = (samples[i].getPosition() - nbrs[j].getSample()->getPosition()).norm();
305  nbrs[j].setSeparation(nbr_sep);
306  }
307  }
308 
309  samples[i].getNeighbors().setCapacity(options.max_degree);
310  for (size_t j = 0; j < nbrs.size(); ++j)
311  samples[i].getNeighbors().insert(nbrs[j]);
312  }
313 
314  updateAverageSeparation();
315  valid_graph = true;
316 
317  return true;
318 }

320 bool
321 PointSet3::save(std::string const & graph_path, std::string const & samples_path, bool write_distances) const
322 {
323  // Save graph
324  std::ofstream gout(graph_path.c_str(), std::ios::binary);
325  if (!gout)
326  {
327  THEA_ERROR << "PointSet3: Could not open graph file '" << graph_path << "' for writing";
328  return false;
329  }
330 
331  gout << options.max_degree << '\n';
332 
333  for (size_t i = 0; i < samples.size(); ++i)
334  {
335  auto const & nbrs = samples[i].getNeighbors();
336  gout << nbrs.size();
337 
338  for (int j = 0; j < nbrs.size(); ++j)
339  gout << ' ' << nbrs[j].getSample()->getIndex();
340 
341  if (write_distances)
342  {
343  for (int j = 0; j < nbrs.size(); ++j)
344  gout << ' ' << nbrs[j].getSeparation();
345  }
346 
347  gout << '\n';
348  }
349 
350  // Save samples
351  if (!samples_path.empty())
352  {
353  std::ofstream sout(samples_path.c_str(), std::ios::binary);
354  if (!sout)
355  {
356  THEA_ERROR << "PointSet3: Could not open samples file '" << samples_path << "' for writing";
357  return false;
358  }
359 
360  for (size_t i = 0; i < samples.size(); ++i)
361  {
362  Vector3 const & p = samples[i].getPosition();
363  sout << p[0] << ' ' << p[1] << ' ' << p[2];
364  if (has_normals)
365  {
366  Vector3 const & n = samples[i].getNormal();
367  sout << ' ' << n[0] << ' ' << n[1] << ' ' << n[2];
368  }
369 
370  sout << '\n';
371  }
372  }
373 
374  return true;
375 }
