216 {

247 intx
248 DisplayMesh::addFace(int num_vertices, intx const * vertex_indices, intx src_face_index)
249 {
250  if (num_vertices < 3)
251  {
252  THEA_DEBUG << getName() << ": Skipping face -- too few vertices (" << num_vertices << ')';
253  return -1;
254  }
255 
256  if (num_vertices == 3)
257  return addTriangle(vertex_indices[0], vertex_indices[1], vertex_indices[2], src_face_index);
258 
259 #ifdef THEA_DEBUG
260  for (int i = 0; i < num_vertices; ++i)
261  {
262  intx vi = vertex_indices[i];
263  debugAssertM(vi >= 0 && vi < (intx)vertices.size(), getName() + format(": Vertex index %ld out of bounds", vi));
264  }
265 #endif
266 
267  intx num_tris = 0;
268  if (num_vertices == 4)
269  {
270  triangulated_indices.resize(6); // std::vector spec minimizes reallocs by never shrinking storage
271  num_tris = Polygon3::triangulateQuad(vertices[(size_t)vertex_indices[0]],
272  vertices[(size_t)vertex_indices[1]],
273  vertices[(size_t)vertex_indices[2]],
274  vertices[(size_t)vertex_indices[3]],
275  triangulated_indices[0], triangulated_indices[1], triangulated_indices[2],
276  triangulated_indices[3], triangulated_indices[4], triangulated_indices[5]);
277  }
278  else
279  {
280  Polygon3 poly;
281  for (int i = 0; i < num_vertices; ++i)
282  poly.addVertex(vertices[(size_t)vertex_indices[i]]);
283 
284  num_tris = poly.triangulate(triangulated_indices);
285  }
286 
287  if (num_tris <= 0)
288  {
289  THEA_DEBUG << getName() << ": Skipping degenerate face -- no triangles produced";
290  return -1;
291  }
292 
293  // debugAssertM(num_tris == 3 * (num_vertices - 2),
294  // getName() + format(": Triangulation of polygonal face yielded %l triangles, whereas %l were expected",
295  // num_tris, num_vertices - 2));
296 
297  alwaysAssertM((src_face_index >= 0 && 3 * tri_src_face_indices.size() == tris.size())
298  || (src_face_index < 0 && tri_src_face_indices.empty()),
299  getNameStr() + ": Mesh must have all or no triangle face source indices");
300 
301 #ifdef THEA_DEBUG
302  intx first_tri = numTriangles();
303  debugAssertM(face_starting_tris.back() == first_tri, getNameStr() + ": Last entry of face_starting_tris is invalid");
304 #endif
305 
306  size_t num_tri_verts = (size_t)(3 * num_tris);
307  for (size_t i = 0; i < num_tri_verts; i += 3)
308  {
309  tris.push_back((uint32)vertex_indices[triangulated_indices[i ]]);
310  tris.push_back((uint32)vertex_indices[triangulated_indices[i + 1]]);
311  tris.push_back((uint32)vertex_indices[triangulated_indices[i + 2]]);
312  if (src_face_index >= 0) tri_src_face_indices.push_back(src_face_index);
313  }
314 
315  intx face_index = numFaces();
316  face_starting_tris.push_back(numTriangles()); // old final entry of face_starting_tris is index of first new triangle
317 
318  if (num_vertices >= 5 && num_tris >= 2) // bona fide higher-degree poly which does not reduce to a single triangle
319  large_poly_verts[first_tri].assign(vertex_indices, vertex_indices + num_vertices);
320 
321  // Add edges of original polygon
322  {
323  uint32 prev = (uint32)vertex_indices[num_vertices - 1];
324  for (int j = 0; j < num_vertices; ++j)
325  {
326  uint32 curr = (uint32)vertex_indices[j];
327  addEdge(prev, curr);
328  prev = curr;
329  }
330  }
331 
332  invalidateGpuBuffers(BufferId::TOPOLOGY);
333 
334  return face_index;
335 }

337 DisplayMesh::Face
338 DisplayMesh::getFace(intx face_index)
339 {
340  if (face_index < 0 || face_index + 1 >= (intx)face_starting_tris.size())
341  return Face();
342 
343  auto begin = face_starting_tris[(size_t)face_index], end = face_starting_tris[(size_t)face_index + 1];
344  auto num_tris = end - begin;
345 
346  switch (num_tris)
347  {
348  case 1: return Face(this, 3, begin, 1); // simple triangle
349 
350  case 2: // actual quad or larger degenerate poly which only produced two valid triangles
351  {
352  auto loc = large_poly_verts.find(begin);
353  return Face(this, (loc == large_poly_verts.end() ? 4 : (int)loc->second.size()), begin, 2);
354  }
355 
356  case 0: return Face(); // totally degenerate, produced no triangles
357 
358  default: // higher-degree polygon
359  {
360  auto loc = large_poly_verts.find(begin);
361  debugAssertM(loc != large_poly_verts.end(), getNameStr() + format(": Vertices of face %ld not found", (long)face_index));
362  return Face(this, (int)loc->second.size(), begin, num_tris);
363  }
364  }

367 void
368 DisplayMesh::computeAveragedVertexNormals()
369 {
370  bool topo_change = (normals.size() != vertices.size());
371 
372  normals.resize(vertices.size());
373  for (size_t i = 0; i < normals.size(); ++i)
374  normals[i] = Vector3::Zero();
375 
376  // TODO: weight normals by face area?
377  for (size_t i = 0; i < tris.size(); i += 3)
378  {
379  uint32 i0 = tris[i], i1 = tris[i + 1], i2 = tris[i + 2];
380  Vector3 const & v0 = vertices[i0];
381  Vector3 const & v1 = vertices[i1];
382  Vector3 const & v2 = vertices[i2];
383 
384  Vector3 n = (v2 - v1).cross(v0 - v1).normalized();
385  normals[i0] += n;
386  normals[i1] += n;
387  normals[i2] += n;
388  }
389 
390  for (size_t i = 0; i < normals.size(); ++i)
391  normals[i] = normals[i].normalized();
392 
393  invalidateGpuBuffers((topo_change ? BufferId::TOPOLOGY : 0) | BufferId::VERTEX_NORMAL);
394 }

405 void
406 DisplayMesh::recomputeEdges()
407 {
408  edges.clear();
409 
410  intx num_faces = numFaces();
411  for (intx i = 0; i < num_faces; ++i)
412  {
413  if (isTriangle(i))
414  {
415  intx tri_base = 3 * face_starting_tris[(size_t)i];
416  for (uint32 j = 0; j < 3; ++j)
417  addEdge(tris[tri_base + j], tris[tri_base + (j + 1) % 3]);
418  }
419  else
420  {
421  auto f = getFace(i);
422  if (!f) continue;
423 
424  auto nv = f.numVertices();
425  if (nv <= 0) continue;
426 
427  uint32 ei0 = f.getVertexIndex(nv - 1);
428  for (int j = 0; j < nv; ++j)
429  {
430  uint32 ei1 = f.getVertexIndex(j);
431  addEdge(ei0, ei1);
432  ei0 = ei1;
433  }
434  }
435  }
436 
437  invalidateGpuBuffers(BufferId::TOPOLOGY);
438 
439  THEA_DEBUG << getName() << ": Mesh has " << edges.size() / 2 << " edges";
440 }

454 void
455 DisplayMesh::isolateTriangles()
456 {
457  // Note: does not change face_starting_tris
458 
459  VertexArray new_vertices;
460  NormalArray new_normals;
461  ColorArray new_colors;
462  TexCoordArray new_texcoords;
463 
464  for (size_t i = 0; i < tris.size(); i += 3)
465  {
466  size_t i0 = (size_t)tris[i ];
467  size_t i1 = (size_t)tris[i + 1];
468  size_t i2 = (size_t)tris[i + 2];
469 
470  uint32 new_vindex = (uint32)new_vertices.size();
471 
472  new_vertices.push_back(vertices[i0]);
473  new_vertices.push_back(vertices[i1]);
474  new_vertices.push_back(vertices[i2]);
475 
476  if (!normals.empty())
477  {
478  new_normals.push_back(normals[i0]);
479  new_normals.push_back(normals[i1]);
480  new_normals.push_back(normals[i2]);
481  }
482 
483  if (!colors.empty())
484  {
485  new_colors.push_back(colors[i0]);
486  new_colors.push_back(colors[i1]);
487  new_colors.push_back(colors[i2]);
488  }
489 
490  if (!texcoords.empty())
491  {
492  new_texcoords.push_back(texcoords[i0]);
493  new_texcoords.push_back(texcoords[i1]);
494  new_texcoords.push_back(texcoords[i2]);
495  }
496 
497  tris[i ] = new_vindex;
498  tris[i + 1] = new_vindex + 1;
499  tris[i + 2] = new_vindex + 2;
500  }
501 
502  vertices = new_vertices;
503  normals = new_normals;
504  colors = new_colors;
505  texcoords = new_texcoords;
506 
507  recomputeEdges();
508 
509  invalidateGpuBuffers();
510 }

524 bool
525 DisplayMesh::uploadToGraphicsSystem(IRenderSystem & render_system, IRenderOptions const & options)
526 {
527  if (!isGpuBufferValid(BufferId::TOPOLOGY) || ((bool)options.drawEdges() == packed_edges.empty() && !edges.empty()))
528  invalidateGpuBuffers(BufferId::ALL); // need to reallocate pool
529 
530  if (changed_buffers == 0) return true;
531 
532  if (changed_buffers == BufferId::ALL)
533  {
534  if (buf_pool) buf_pool->reset();
535  vertices_buf = normals_buf = colors_buf = texcoords_buf = tris_buf = edges_buf = nullptr;
536 
537  if (vertices.empty() || (tris.empty() && edges.empty()))
538  {
539  if (buf_pool)
540  {
541  render_system.destroyBufferPool(buf_pool);
542  buf_pool = nullptr;
543  }
544 
545  setAllGpuBuffersValid();
546  return true;
547  }
548 
549  static int const PADDING = 32;
550  intx vertex_bytes = !vertices.empty() ? 3 * 4 * (intx)vertices.size() + PADDING : 0; // 3 * float
551  intx normal_bytes = hasNormals() ? 3 * 4 * (intx)normals.size() + PADDING : 0; // 3 * float
552  intx color_bytes = hasColors() ? 4 * 4 * (intx)colors.size() + PADDING : 0; // 4 * float
553  intx texcoord_bytes = hasTexCoords() ? 2 * 4 * (intx)texcoords.size() + PADDING : 0; // 2 * float
554 
555  // Create the edge buffer only if we need it, else save GPU memory. We assume toggling the edge display on/off will be a
556  // relatively rare occurrence.
557  if (options.drawEdges()) packEdges();
558  else packed_edges.clear();
559 
560  // After this, options.drawEdges() != packed_edges.empty() except in the trivial case of edges.empty().
561 
562 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
563  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + PADDING;
564 #else
565  intx tri_bytes = !tris.empty() ? 4 * (intx)tris.size() + PADDING : 0; // uint32
566  intx edge_bytes = !packed_edges.empty() ? 4 * (intx)packed_edges.size() + PADDING : 0; // uint32
567 
568  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + tri_bytes + edge_bytes + PADDING;
569 #endif
570 
571  if (buf_pool)
572  {
573  if (buf_pool->getCapacity() <= num_bytes || buf_pool->getCapacity() > (intx)(1.5 * num_bytes))
574  {
575  render_system.destroyBufferPool(buf_pool);
576 
577  std::string pool_name = getNameStr() + " buffer pool";
578  buf_pool = render_system.createBufferPool(pool_name.c_str(), num_bytes, IBufferPool::Usage::WRITE_OCCASIONALLY, true);
579  if (!buf_pool) return false;
580  }
581  // Else no need to reset buf_pool, we've done it above
582  }
583  else
584  {
585  std::string pool_name = getNameStr() + " buffer pool";
586  buf_pool = render_system.createBufferPool(pool_name.c_str(), num_bytes, IBufferPool::Usage::WRITE_OCCASIONALLY, true);
587  if (!buf_pool) return false;
588  }
589 
590  if (!vertices.empty() && !(vertices_buf = buf_pool->createBuffer(vertex_bytes)) ) return false;
591  if (hasNormals() && !(normals_buf = buf_pool->createBuffer(normal_bytes)) ) return false;
592  if (hasColors() && !(colors_buf = buf_pool->createBuffer(color_bytes)) ) return false;
593  if (hasTexCoords() && !(texcoords_buf = buf_pool->createBuffer(texcoord_bytes))) return false;
594 
595 #ifndef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
596  if (!tris.empty() && !(tris_buf = buf_pool->createBuffer(tri_bytes))) return false;
597  if (!packed_edges.empty() && !(edges_buf = buf_pool->createBuffer(edge_bytes))) return false;
598 
599  if (!tris.empty() && !tris_buf->updateIndices (0, (int64)tris.size(), NumericType::UINT32, &tris[0]))
600  return false;
601 
602  if (!packed_edges.empty()
603  && !edges_buf->updateIndices(0, (int64)packed_edges.size(), NumericType::UINT32, &packed_edges[0]))
604  return false;
605 #endif
606 
607  if (!vertices.empty() && !vertices_buf->updateAttributes(0, (int64)vertices.size(), 3, NumericType::REAL, &vertices[0]))
608  return false;
609 
610  if (hasNormals() && !normals_buf->updateAttributes(0, (int64)normals.size(), 3, NumericType::REAL, &normals[0]))
611  return false;
612 
613  if (hasColors() && !colors_buf->updateAttributes(0, (int64)colors.size(), 4, NumericType::REAL, &colors[0]))
614  return false;
615 
616  if (hasTexCoords() && !texcoords_buf->updateAttributes(0, (int64)texcoords.size(), 2, NumericType::REAL, &texcoords[0]))
617  return false;
618  }
619  else
620  {
621  if (!isGpuBufferValid(BufferId::VERTEX_POSITION) && !vertices.empty()

641 {
642  if (!render_system) { THEA_ERROR << getName() << ": Can't display mesh on a null rendersystem"; return false; }
643  if (!options) options = RenderOptions::defaults();
644 
645  if (!const_cast<DisplayMesh *>(this)->uploadToGraphicsSystem(*render_system, *options)) return false;
646 
647  if (!vertices_buf) return true;
648  if (!options->drawFaces() && !options->drawEdges()) return true;
649  if (!options->drawFaces() && edges.empty()) return true;
650  if (!options->drawEdges() && tris.empty()) return true;
651 
652  render_system->beginIndexedPrimitives();
653 
654  render_system->setVertexBuffer(vertices_buf);
655  if (options->sendNormals() && normals_buf) render_system->setNormalBuffer(normals_buf);
656  if (options->sendColors() && colors_buf) render_system->setColorBuffer(colors_buf);
657  if (options->sendTexCoords() && texcoords_buf) render_system->setTexCoordBuffer(0, texcoords_buf);
658 
659  if (options->drawFaces())
660  {
661  if (options->drawEdges())
662  {
663  render_system->pushShapeFlags();
664  render_system->setPolygonOffset(true, 1);
665  }
666 
667  if (!tris.empty())
668  {
669 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
670  render_system->sendIndices(IRenderSystem::Primitive::TRIANGLES, (int64)tris.size(), &tris[0]);
671 #else
672  render_system->setIndexBuffer(tris_buf);
673  render_system->sendIndicesFromBuffer(IRenderSystem::Primitive::TRIANGLES, 0, (int64)tris.size());
674 #endif
675  }
676 
677  if (options->drawEdges())
678  render_system->popShapeFlags();
679  }
680 
681  if (options->drawEdges())
682  {
683  render_system->pushShader();
684  render_system->pushColorFlags();
685  render_system->pushTextures();
686 
687  render_system->setShader(nullptr);
688  render_system->setColorBuffer(nullptr);
689  render_system->setTexCoordBuffer(0, nullptr);
690  render_system->setNormalBuffer(nullptr);
691  render_system->setColor(options->edgeColor()); // set default edge color (TODO: handle per-edge colors)
692  render_system->setTexture(0, nullptr);
693 
694  if (!packed_edges.empty())
695  {
696 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
697  render_system->sendIndices(IRenderSystem::Primitive::LINES, (int64)packed_edges.size(), &packed_edges[0]);
698 #else
699  render_system->setIndexBuffer(edges_buf);
700  render_system->sendIndicesFromBuffer(IRenderSystem::Primitive::LINES, 0, (int64)packed_edges.size());
701 #endif
702  }
703 
704  render_system->popTextures();
705  render_system->popColorFlags();
706  render_system->popShader();
707  }
708 
709  render_system->endIndexedPrimitives();
710 
711  if (char const * err = render_system->getLastError())
712  { THEA_ERROR << getName() << ": Rendering error (" << err << ')'; return false; }
713
