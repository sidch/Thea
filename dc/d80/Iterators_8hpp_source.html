115 #define THEA_RANDOM_ACCESS_ITERATOR_BODY(class_name, difference_type) \
116  class_name(class_name const & src) : ii(src.ii) {} \
117  \
118  class_name & operator++() { ++ii; return *this; } \
119  class_name operator++(int) { class_name tmp(*this); operator++(); return tmp; } \
120  class_name & operator--() { --ii; return *this; } \
121  class_name operator--(int) { class_name tmp(*this); operator--(); return tmp; } \
122  \
123  class_name & operator=(class_name const & src) { ii = src.ii; return *this; } \
124  class_name & operator+=(difference_type n) { ii += n; return *this; } \
125  class_name & operator-=(difference_type n) { ii -= n; return *this; } \
126  \
127  class_name operator+(difference_type n) const { return class_name(ii + n); } \
128  class_name operator-(difference_type n) const { return class_name(ii - n); } \
129  difference_type operator-(class_name const & rhs) const { return ii - rhs.ii; } \
130  \
131  bool operator==(class_name const & rhs) const { return ii == rhs.ii; } \
132  bool operator!=(class_name const & rhs) const { return ii != rhs.ii; } \
133  \
134  bool operator< (class_name const & rhs) const { return ii < rhs.ii; } \
135  bool operator> (class_name const & rhs) const { return ii > rhs.ii; } \
136  bool operator<=(class_name const & rhs) const { return ii <= rhs.ii; } \
137  bool operator>=(class_name const & rhs) const { return ii >= rhs.ii; }
