32 // Do a random walk upto \a num_steps steps, and return the number of steps actually taken (= \a num_steps except in corner
33 // cases).
34 intx
35 walk(PointSet3 const & surf, intx seed_index, intx num_steps, double * features)
36 {
37  SamplePoint3 const * sample = &surf.getSample(seed_index);
38  intx base_index = 0;
39  for (intx i = 0; i < num_steps; ++i, base_index += 3)
40  {
41  auto const & nbrs = sample->getNeighbors();
42  if (nbrs.empty())
43  return i;
44 
45  intx next_index = Random::common().integer(0, nbrs.size() - 1);
46  sample = nbrs[next_index].getSample();
47 
48  Vector3 const & p = sample->getPosition();
49  features[base_index ] += p[0];
50  features[base_index + 1] += p[1];
51  features[base_index + 2] += p[2];
52  }
53 
54  return num_steps;
55 }

68 void
69 RandomWalks::compute(Vector3 const & position, intx num_steps, double * features, intx num_walks) const
70 {
71  static intx const DEFAULT_NUM_WALKS = 1000;
72 
73  if (num_walks < 0)
74  num_walks = DEFAULT_NUM_WALKS;
75 
76  // Zero out all features initially
77  for (intx i = 0; i < num_steps; ++i)
78  {
79  features[3 * i ] = 0.0;
80  features[3 * i + 1] = 0.0;
81  features[3 * i + 2] = 0.0;
82  }
83 
84  // Find the sample closest to the query position and use it as the source for walks
85  intx seed_index = surf->getBvh().closestElement<MetricL2>(position);
86  alwaysAssertM(seed_index >= 0, "RandomWalks: Seed sample for random walks not found");
87 
88  // Make sure the proximity graph exists, since we don't actually call getGraph()
89  surf->updateGraph();
90 
91  // Do the walks
92  Array<intx> counts((size_t)num_steps, 0);
93  for (intx i = 0; i < num_walks; ++i)
94  {
95  intx walk_steps = RandomWalksInternal::walk(*surf, seed_index, num_steps, features);
96  for (intx j = 0; j < walk_steps; ++j)
97  counts[(size_t)j]++;
98  }
99 
100  // Compute average offsets
101  for (intx i = 0; i < num_steps; ++i)
102  {
103  intx count = counts[(size_t)i];
104  features[3 * i ] = (features[3 * i ] / count - position[0]);
105  features[3 * i + 1] = (features[3 * i + 1] / count - position[1]);
106  features[3 * i + 2] = (features[3 * i + 2] / count - position[2]);
107  }
108 }
