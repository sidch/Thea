243 // Dirty SFINAE template trick to check if the type T has get...() and set...() functions
244 #define THEA_HAS_GRAPHICS_ATTRIB(name, get_func_name, set_func_name) \
245 template <typename T> \
246 class name \
247 { \
248  public: \
249  typedef char Yes; \
250  typedef struct { char a[2]; } No; \
251 \
252  struct GetBaseMixin { void get_func_name() {} }; \
253  struct SetBaseMixin { void set_func_name() {} }; \
254 \
255  struct GetBase : public T, public GetBaseMixin {}; \
256  struct SetBase : public T, public SetBaseMixin {}; \
257  template <typename S, S> class Helper {}; \
258 \
259  template <typename U> static No testGet(U *, Helper<void (GetBaseMixin::*)(), &U::get_func_name> * = 0); \
260  static Yes testGet(...); \
261 \
262  template <typename U> static No testSet(U *, Helper<void (SetBaseMixin::*)(), &U::set_func_name> * = 0); \
263  static Yes testSet(...); \
264 \
265  public: \
266  static bool const has_get = (sizeof(testGet((GetBase *)(0))) == sizeof(Yes)); \
267  static bool const has_set = (sizeof(testSet((SetBase *)(0))) == sizeof(Yes)); \
268  static bool const value = has_get && has_set; \
269 };
