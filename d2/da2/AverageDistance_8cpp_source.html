62 double
63 computeEuclidean(PointSet3 const & surf, Vector3 const & position, Real max_distance)
64 {
65  bool process_all = (max_distance < 0);
66  if (process_all)
67  max_distance = surf.getScale();
68 
69  EuclideanCallback callback(position);
70 
71  if (process_all)
72  {
73  intx num_samples = surf.numSamples();
74  for (intx i = 0; i < num_samples; ++i)
75  callback(i, surf.getSample(i).getPosition());
76  }
77  else
78  {
79  Ball3 ball(position, max_distance);
80  const_cast<PointSet3::SampleBvh &>(surf.getBvh()).processRangeUntil<IntersectionTester>(ball, std::ref(callback));
81  }
82 
83  return callback.getAverageDistance() / max_distance;
84 }

110 double
111 computeGeodesic(PointSet3 const & surf, Vector3 const & position, Real max_distance)
112 {
113  bool process_all = (max_distance < 0);
114  if (process_all)
115  max_distance = surf.getScale();
116 
117  // Find the sample closest to the query position and use it as the source for all distance calculations
118  intx seed_index = surf.getBvh().closestElement<MetricL2>(position);
119  alwaysAssertM(seed_index >= 0, "AverageDistance: Seed sample for geodesic distances not found");
120 
121  // Assume the graph and the BVH have samples in the same sequence
122  PointSet3::SampleGraph::VertexHandle seed_sample = const_cast<SamplePoint3 *>(&surf.getSample(seed_index));
123 
124  ShortestPaths<PointSet3::SampleGraph> shortest_paths;
125  GeodesicCallback callback;
126  shortest_paths.dijkstraWithCallback(const_cast<PointSet3::SampleGraph &>(surf.getGraph()), seed_sample, std::ref(callback),
127  (process_all ? -1 : max_distance));
128 
129  return callback.getAverageDistance() / max_distance;
130 }
